class BankersAlgorithm {
    private $processes;
    private $resources;
    private $available;
    private $max;
    private $allocation;
    private $need;

    /**
     * Constructor for BankersAlgorithm class
     * @param int $processes Number of processes
     * @param int $resources Number of resource types
     */
    public function __construct($processes, $resources) {
        $this->processes = $processes;
        $this->resources = $resources;
        $this->available = array_fill(0, $resources, 0);
        $this->max = array_fill(0, $processes, array_fill(0, $resources, 0));
        $this->allocation = array_fill(0, $processes, array_fill(0, $resources, 0));
        $this->need = array_fill(0, $processes, array_fill(0, $resources, 0));
    }

    /**
     * Sets available resources
     * @param array $available Array of available resources
     */
    public function setAvailable($available) {
        if (count($available) !== $this->resources) {
            throw new Exception("Invalid number of resources");
        }
        $this->available = $available;
    }

    /**
     * Sets maximum resources for each process
     * @param array $max 2D array of maximum resources
     */
    public function setMax($max) {
        if (count($max) !== $this->processes) {
            throw new Exception("Invalid number of processes");
        }
        foreach ($max as $process) {
            if (count($process) !== $this->resources) {
                throw new Exception("Invalid number of resources");
            }
        }
        $this->max = $max;
    }

    /**
     * Sets allocated resources for each process
     * @param array $allocation 2D array of allocated resources
     */
    public function setAllocation($allocation) {
        if (count($allocation) !== $this->processes) {
            throw new Exception("Invalid number of processes");
        }
        foreach ($allocation as $process) {
            if (count($process) !== $this->resources) {
                throw new Exception("Invalid number of resources");
            }
        }
        $this->allocation = $allocation;

        // Calculate need matrix
        for ($i = 0; $i < $this->processes; $i++) {
            for ($j = 0; $j < $this->resources; $j++) {
                $this->need[$i][$j] = $this->max[$i][$j] - $this->allocation[$i][$j];
                
                // Reduce available resources
                if ($i === 0) { // Only subtract once per resource type
                    $this->available[$j] -= $allocation[$i][$j];
                }
            }
        }
    }

    /**
     * Checks if the system is in safe state
     * @return array|bool Returns safe sequence if safe, false otherwise
     */
    public function isSafe() {
        $work = $this->available;
        $finish = array_fill(0, $this->processes, false);
        $safeSeq = array_fill(0, $this->processes, 0);
        
        $count = 0;
        while ($count < $this->processes) {
            $found = false;
            for ($i = 0; $i < $this->processes; $i++) {
                if (!$finish[$i]) {
                    $possible = true;
                    for ($j = 0; $j < $this->resources; $j++) {
                        if ($this->need[$i][$j] > $work[$j]) {
                            $possible = false;
                            break;
                        }
                    }
                    
                    if ($possible) {
                        for ($j = 0; $j < $this->resources; $j++) {
                            $work[$j] += $this->allocation[$i][$j];
                        }
                        
                        $safeSeq[$count] = $i;
                        $finish[$i] = true;
                        $count++;
                        $found = true;
                    }
                }
            }
            
            if (!$found) {
                return false;
            }
        }
        
        return $safeSeq;
    }

    /**
     * Handles resource request from a process
     * @param int $pid Process ID
     * @param array $request Resource request array
     * @return bool True if request can be granted, false otherwise
     */
    public function requestResources($pid, $request) {
        $result = [
            'status' => false,
            'message' => '',
            'safe_sequence' => []
        ];

        // Check if request exceeds maximum claim
        for ($i = 0; $i < $this->resources; $i++) {
            if ($request[$i] > $this->max[$pid][$i]) {
                $result['message'] = "Error: Request exceeds maximum claim for process P$pid";
                return $result;
            }
        }
        
        // Check if resources are available
        for ($i = 0; $i < $this->resources; $i++) {
            if ($request[$i] > $this->available[$i]) {
                $result['message'] = "Resources not available for process P$pid";
                return $result;
            }
        }
        
        // Try to allocate resources temporarily
        $tempAvailable = $this->available;
        $tempAllocation = $this->allocation;
        $tempNeed = $this->need;
        
        for ($i = 0; $i < $this->resources; $i++) {
            $tempAvailable[$i] -= $request[$i];
            $tempAllocation[$pid][$i] += $request[$i];
            $tempNeed[$pid][$i] -= $request[$i];
        }
        
        // Save current state
        $saveAvailable = $this->available;
        $saveAllocation = $this->allocation;
        $saveNeed = $this->need;
        
        // Set temporary state
        $this->available = $tempAvailable;
        $this->allocation = $tempAllocation;
        $this->need = $tempNeed;
        
        // Check if the state is safe
        $safeSeq = $this->isSafe();
        
        if ($safeSeq !== false) {
            // If safe, keep the changes
            $result['status'] = true;
            $result['message'] = "Resources allocated to process P$pid successfully";
            $result['safe_sequence'] = $safeSeq;
        } else {
            // If not safe, restore previous state
            $this->available = $saveAvailable;
            $this->allocation = $saveAllocation;
            $this->need = $saveNeed;
            $result['message'] = "Resource allocation to process P$pid denied (would lead to unsafe state)";
        }
        
        return $result;
    }

    /**
     * Gets the current state of the system
     * @return array System state
     */
    public function getState() {
        return [
            'processes' => $this->processes,
            'resources' => $this->resources,
            'available' => $this->available,
            'max' => $this->max,
            'allocation' => $this->allocation,
            'need' => $this->need
        ];
    }
}

// Handle AJAX requests
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    header('Content-Type: application/json');
    
    $action = $_POST['action'] ?? '';
    
    switch($action) {
        case 'initialize':
            $processes = intval($_POST['processes'] ?? 0);
            $resources = intval($_POST['resources'] ?? 0);
            
            if ($processes <= 0 || $resources <= 0) {
                echo json_encode(['status' => 'error', 'message' => 'Invalid number of processes or resources']);
                exit;
            }
            
            // Create and store the banker's algorithm object in session
            session_start();
            $_SESSION['bankers'] = new BankersAlgorithm($processes, $resources);
            
            echo json_encode([
                'status' => 'success', 
                'message' => "Initialized with $processes processes and $resources resource types"
            ]);
            break;
            
        case 'set_resources':
            session_start();
            if (!isset($_SESSION['bankers'])) {
                echo json_encode(['status' => 'error', 'message' => 'System not initialized']);
                exit;
            }
            
            $available = json_decode($_POST['available'] ?? '[]', true);
            $max = json_decode($_POST['max'] ?? '[]', true);
            $allocation = json_decode($_POST['allocation'] ?? '[]', true);
            
            try {
                $_SESSION['bankers']->setAvailable($available);
                $_SESSION['bankers']->setMax($max);
                $_SESSION['bankers']->setAllocation($allocation);
                
                $safeSeq = $_SESSION['bankers']->isSafe();
                
                if ($safeSeq !== false) {
                    echo json_encode([
                        'status' => 'success',
                        'message' => 'System is in safe state',
                        'safe_sequence' => $safeSeq,
                        'state' => $_SESSION['bankers']->getState()
                    ]);
                } else {
                    echo json_encode([
                        'status' => 'warning',
                        'message' => 'System is not in safe state',
                        'state' => $_SESSION['bankers']->getState()
                    ]);
                }
            } catch (Exception $e) {
                echo json_encode(['status' => 'error', 'message' => $e->getMessage()]);
            }
            break;
            
        case 'request_resources':
            session_start();
            if (!isset($_SESSION['bankers'])) {
                echo json_encode(['status' => 'error', 'message' => 'System not initialized']);
                exit;
            }
            
            $pid = intval($_POST['pid'] ?? -1);
            $request = json_decode($_POST['request'] ?? '[]', true);
            
            $state = $_SESSION['bankers']->getState();
            if ($pid < 0 || $pid >= $state['processes']) {
                echo json_encode(['status' => 'error', 'message' => 'Invalid process ID']);
                exit;
            }
            
            $result = $_SESSION['bankers']->requestResources($pid, $request);
            
            if ($result['status']) {
                echo json_encode([
                    'status' => 'success',
                    'message' => $result['message'],
                    'safe_sequence' => $result['safe_sequence'],
                    'state' => $_SESSION['bankers']->getState()
                ]);
            } else {
                echo json_encode([
                    'status' => 'warning',
                    'message' => $result['message'],
                    'state' => $_SESSION['bankers']->getState()
                ]);
            }
            break;
            
        default:
            echo json_encode(['status' => 'error', 'message' => 'Invalid action']);
    }
    
    exit;
}

Knowledge

Learn the Banker's Algorithm

What is the Banker's Algorithm?

The Banker's Algorithm is a deadlock avoidance algorithm used in operating systems. It was developed by Edsger Dijkstra and is named the "Banker's algorithm" because it models how a bank would never allocate its available cash in a way that would prevent it from fulfilling all customer needs.

How does it work?

The algorithm maintains information about resource allocation to each process, the maximum demand of each process, and the available resources. It uses this information to determine if granting a resource request would put the system in an unsafe state.

Key Components:

📦
Available Resources:
The number of available instances of each resource type.
📊
Maximum Need:
The maximum demand of each process for each resource type.
🔧
Allocation:
The number of resources of each type allocated to each process.
❓
Need:
The remaining resources needed by each process (Max - Allocation).

Algorithm Steps:

  1. 1Calculate the Need matrix by subtracting the Allocation matrix from the Maximum matrix.
  2. 2Look for a process whose needs can be satisfied with the available resources.
  3. 3If found, temporarily add its allocated resources to the available resources.
  4. 4Repeat steps 2-3 until all processes are allocated or no process can be allocated.
  5. 5If all processes can complete, the system is in a safe state.
